    <input type="text" ng-model="newValue"/>
    <section data-ng-repeat="value in array | filter:newValue">{{value}};</section>
      
    <p><strong>built in UPPERCASE filter</strong></p>
    <section data-ng-repeat="value in array">{{value | uppercase}};</section>
    
    <p><strong>custom REVERSE filter</strong></p>
    <section data-ng-repeat="value in array">{{value | reverse}};</section>
  </div>


  // Just put the model into the scope and you're good to go
  app.controller('filterCtrl', function($scope){
    $scope.array = ['cat one', 'cat two', 'cat three', 'dog one', 'dog two', 'dog three'];
  });
  
  // Define your own filters easily
  app.filter('reverse', function() {
    return function(input) {
      var out = "";
      for (var i = 0; i < input.length; i++) {
        out = input.charAt(i) + out;
      }
      return out;
    }
  });
