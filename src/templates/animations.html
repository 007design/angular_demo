  function toggleSection(id) {
    var isVisible = $('#'+id).is(':visible');
    if (isVisible)
      $('#'+id).animate({ opacity: 0, padding-top: 100px });
    else
      $('#'+id).animate({ opacity: 1, padding-top: 0px });
  }
